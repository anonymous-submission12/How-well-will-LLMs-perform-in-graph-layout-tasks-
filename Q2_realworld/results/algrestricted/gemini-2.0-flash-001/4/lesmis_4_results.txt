def community_detection(edge_list):
    """
    Detects communities in a graph using a simple label propagation algorithm.
    """
    # Build the graph as an adjacency list
    graph = defaultdict(list)
    for u, v in edge_list:
        graph[u].append(v)
        graph[v].append(u)

    # Initialize each node with a unique label (its own ID)
    labels = {node: node for node in graph}

    # Iterate until labels stabilize
    changed = True
    while changed:
        changed = False
        nodes = list(graph.keys())
        random.shuffle(nodes)  # Randomize node order for each iteration

        for node in nodes:
            neighbor_labels = defaultdict(int)
            for neighbor in graph[node]:
                neighbor_labels[labels[neighbor]] += 1

            # Find the most frequent label among neighbors
            if neighbor_labels:
                most_frequent_label = max(neighbor_labels, key=neighbor_labels.get)

                # Update the node's label if it's different from the most frequent neighbor label
                if labels[node] != most_frequent_label:
                    labels[node] = most_frequent_label
                    changed = True

    # Group nodes by community label
    communities = defaultdict(list)
    for node, label in labels.items():
        communities[label].append(node)

    return list(communities.values())
