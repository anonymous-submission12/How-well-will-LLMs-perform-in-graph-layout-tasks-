def calculate_highest_degree(edge_list):
    """
    Calculates the highest degree of all nodes in a graph represented as an edge list.

    Args:
        edge_list: A list of tuples, where each tuple represents an edge in the graph.

    Returns:
        The highest degree of any node in the graph.
    """

    degrees = {}
    for edge in edge_list:
        u, v = edge
        degrees[u] = degrees.get(u, 0) + 1
        degrees[v] = degrees.get(v, 0) + 1

    if not degrees:
        return 0

    return max(degrees.values())
