edge_list = [(0, 1), (0, 7), (1, 2), (1, 8), (2, 3), (2, 9), (3, 4), (3, 10), (4, 5), (4, 11), (5, 6), (5, 12), (6, 13), (7, 8), (7, 14), (8, 9), (8, 15), (9, 10), (9, 16), (10, 11), (10, 17), (11, 12), (11, 18), (12, 13), (12, 19), (13, 20), (14, 15), (15, 16), (16, 17), (17, 18), (18, 19), (19, 20)]

degrees = {}
for u, v in edge_list:
    if u not in degrees:
        degrees[u] = 0
    if v not in degrees:
        degrees[v] = 0
    degrees[u] += 1
    degrees[v] += 1

max_degree = 0
for node in degrees:
    if degrees[node] > max_degree:
        max_degree = degrees[node]

print(max_degree)
