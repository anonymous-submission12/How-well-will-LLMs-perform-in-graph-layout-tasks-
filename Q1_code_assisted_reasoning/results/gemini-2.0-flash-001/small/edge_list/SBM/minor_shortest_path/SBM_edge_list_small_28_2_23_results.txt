def find_shortest_path(graph, start, end):
    adj = {}
    for u, v in graph:
        if u not in adj:
            adj[u] = []
        if v not in adj:
            adj[v] = []
        adj[u].append(v)
        adj[v].append(u)

    queue = [(start, [start])]
    visited = set()

    while queue:
        node, path = queue.pop(0)

        if node == end:
            return len(path) - 1

        visited.add(node)

        if node in adj:
            for neighbor in adj[node]:
                if neighbor not in visited:
                    queue.append((neighbor, path + [neighbor]))
    return float('inf')

shortest_path_length = find_shortest_path(graph, 2, 23)
