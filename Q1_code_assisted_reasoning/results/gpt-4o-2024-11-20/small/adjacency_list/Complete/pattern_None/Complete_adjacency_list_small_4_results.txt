def determine_graph_pattern(adj_list):
    n = len(adj_list)
    all_nodes = set(adj_list.keys())
    
    # Check if the graph is complete
    is_complete = all(len(neighbors) == n - 1 for neighbors in adj_list.values())
    if is_complete:
        return "Complete"

    # Check if the graph is a star
    center_nodes = [node for node, neighbors in adj_list.items() if len(neighbors) == n - 1]
    leaf_nodes = [node for node, neighbors in adj_list.items() if len(neighbors) == 1]
    if len(center_nodes) == 1 and len(leaf_nodes) == n - 1:
        return "Star"

    # Check if the graph is a cycle
    is_cycle = all(len(neighbors) == 2 for neighbors in adj_list.values())
    if is_cycle:
        return "Cycle"

    # Check if the graph is a path
    degree_counts = [len(neighbors) for neighbors in adj_list.values()]
    is_path = degree_counts.count(1) == 2 and degree_counts.count(2) == n - 2
    if is_path:
        return "Path"

    # Check if the graph is a grid
    # A grid graph has nodes with degree 2, 3, or 4, and the number of nodes is a perfect square
    import math
    sqrt_n = int(math.sqrt(n))
    is_grid = sqrt_n * sqrt_n == n and all(2 <= len(neighbors) <= 4 for neighbors in adj_list.values())
    if is_grid:
        return "Grid"

    # Check if the graph is clustered
    # A clustered graph has groups of nodes that are densely connected internally but sparsely connected between groups
    # This is a heuristic check based on average degree
    avg_degree = sum(len(neighbors) for neighbors in adj_list.values()) / n
    if avg_degree < n / 2:
        return "Clustered"

    # If none of the above, assume it's a random graph
    return "Random"
