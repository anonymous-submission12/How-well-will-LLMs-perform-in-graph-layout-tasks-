def determine_graph_pattern(adj_list):
    num_nodes = len(adj_list)
    num_edges = sum(len(neighbors) for neighbors in adj_list.values()) // 2

    # Check for Complete Graph
    if num_edges == num_nodes * (num_nodes - 1) // 2:
        return "Complete"

    # Check for Star Graph
    center_nodes = [node for node, neighbors in adj_list.items() if len(neighbors) == num_nodes - 1]
    leaf_nodes = [node for node, neighbors in adj_list.items() if len(neighbors) == 1]
    if len(center_nodes) == 1 and len(leaf_nodes) == num_nodes - 1:
        return "Star"

    # Check for Cycle Graph
    if all(len(neighbors) == 2 for neighbors in adj_list.values()) and num_edges == num_nodes:
        return "Cycle"

    # Check for Path Graph
    if num_edges == num_nodes - 1 and sum(len(neighbors) == 1 for neighbors in adj_list.values()) == 2 and all(len(neighbors) <= 2 for neighbors in adj_list.values()):
        return "Path"

    # Check for Grid Graph (simplified to detect only rectangular grids)
    from math import sqrt
    possible_side = int(sqrt(num_nodes))
    if possible_side * possible_side == num_nodes:
        is_grid = True
        for node, neighbors in adj_list.items():
            if len(neighbors) not in (2, 3, 4):
                is_grid = False
                break
        if is_grid:
            return "Grid"

    # Check for Clustered Graph
    # (A simple heuristic: more edges than nodes but not fully connected)
    if num_edges > num_nodes and num_edges < num_nodes * (num_nodes - 1) // 2:
        return "Clustered Graph"

    # Random Graph (default case)
    return "Random Graph"
