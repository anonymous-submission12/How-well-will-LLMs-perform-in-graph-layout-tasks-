# Function to calculate the shortest path using BFS
def shortest_path_length(graph, start, target):
    visited = set()
    queue = deque([(start, 0)])  # (current_node, current_distance)
    
    while queue:
        node, distance = queue.popleft()
        if node == target:
            return distance
        if node not in visited:
            visited.add(node)
            for neighbor in graph[node]:
                queue.append((neighbor, distance + 1))
    return -1  # Return -1 if there is no path

# Calculate and print the shortest path length between node '25' and '56'
